CREATE DATABASE Agendamento; 
USE Agendamento;
CREATE TABLE IF NOT EXISTS Paciente(
    Cod_Paci INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
    Nome VARCHAR(200) NOT NULL,
	CPF VARCHAR(14) NOT NULL,  
	Fone VARCHAR(20) NOT NULL,   
	Data_Nasc DATE NOT NULL
); 
CREATE TABLE IF NOT EXISTS Admin(
    Cod_Admin INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
    Nome VARCHAR(200) NOT NULL,
    Registro VARCHAR(11) NOT NULL,
    PASSWORD VARCHAR(200) NOT NULL
); 
CREATE TABLE IF NOT EXISTS Agenda(
    Cod_Agenda INT UNSIGNED NOT NULL PRIMARY KEY AUTO_INCREMENT,
    Dia_Hora DATETIME NOT NULL,
    Descricao VARCHAR(500) NOT NULL,
    Cod_Paci INT NOT NULL, 
    FOREIGN KEY (Cod_Paci) REFERENCES Paciente(Cod_Paci),
    Cod_Admin INT NOT NULL, 
    FOREIGN KEY (Cod_Admin) REFERENCES Admin(Cod_Admin)
);
